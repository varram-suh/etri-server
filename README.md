# install

``` 
sudo apt install qtmultimedia5-dev
sudo apt install libqt5multimedia5-plugins
git clone this-repo
cd this-repo
mkdir build
cd build
cmake ..
make 
```

required boost 1.69 --prefix=/usr/local
